## DESIGN: WHAT SCALES AND HOW

### Difficulty Parameters (6 axes)

| Parameter | Easy (0.1) | Medium (0.5) | Hard (0.9) | Config Key |
|-----------|-----------|-------------|-----------|------------|
| Honeypot density | 5% of nodes | 15% of nodes | 30% of nodes | `ENV_HONEYPOT_DENSITY` |
| Target file count | 1 file | 2 files | 4 files (RED must get all) | `env_target_files` |
| BLUE response speed | 3-step delay | 1-step delay | Instant response | `blue_response_delay` |
| Network complexity | 40 nodes, sparse | 50 nodes, medium | 60 nodes, dense | `ENV_GRAPH_SIZE` |
| Zone lockdown threshold | Suspicion 0.9 triggers | 0.7 triggers | 0.5 triggers | `zone_lockdown_threshold` |
| Trap budget (BLUE) | 3 traps | 5 traps | 8 traps | `ENV_TRAP_BUDGET_BLUE` |

### Escalation Algorithm

```python
class DynamicDifficultyController:
    def __init__(self, window_size: int = 10):
        self.window_size = window_size
        self.history: list[dict] = []  # Last N episode results
        self.current_difficulty = 0.30
        self.axis_weights = {
            "honeypot_density": 0.30,
            "target_files": 0.30,
            "blue_response": 0.50,
            "network_size": 0.50,
            "zone_lockdown": 0.30,
            "trap_budget": 0.50,
        }

    def record_episode(self, result: dict):
        """Record episode outcome for difficulty adjustment."""
        self.history.append({
            "red_won": result["red_reward"] > 0,
            "red_reward": result["red_reward"],
            "steps_used": result["steps"],
            "zones_reached": result["max_zone"],
            "traps_triggered": result["traps_triggered"],
            "emergent_actions": result.get("emergent_count", 0),
        })
        if len(self.history) > self.window_size:
            self.history = self.history[-self.window_size:]

    def compute_next_difficulty(self) -> dict:
        """Multi-axis difficulty adjustment based on performance patterns."""
        if len(self.history) < 3:
            return self._default_params()

        win_rate = sum(1 for h in self.history if h["red_won"]) / len(self.history)
        avg_steps = sum(h["steps_used"] for h in self.history) / len(self.history)
        avg_zones = sum(h["zones_reached"] for h in self.history) / len(self.history)
        avg_traps = sum(h["traps_triggered"] for h in self.history) / len(self.history)

        # Base adjustment from win rate
        if win_rate > 0.6:
            delta = +0.05 * (win_rate - 0.5)  # Harder
        elif win_rate < 0.3:
            delta = -0.02 * (0.4 - win_rate)  # Easier
        else:
            delta = 0.0  # Goldilocks zone

        self.current_difficulty = max(0.10, min(1.0, self.current_difficulty + delta))
        d = self.current_difficulty

        # Per-axis scaling
        return {
            "honeypot_density": 0.05 + d * 0.25,          # 5%–30%
            "target_files": max(1, int(1 + d * 3)),        # 1–4
            "blue_response_delay": max(0, int(3 - d * 3)), # 3–0 steps
            "graph_size": int(40 + d * 20),                 # 40–60 nodes
            "zone_lockdown_threshold": 0.9 - d * 0.4,      # 0.9–0.5
            "trap_budget_blue": int(3 + d * 5),             # 3–8
            "difficulty": d,
            "win_rate_window": win_rate,
        }
```
