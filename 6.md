## OBJECTIVE

Extend the BLUE Forensics agent's capabilities to produce a visual **"crime scene reconstruction"** — a post-episode analysis view that shows judges exactly what RED did, what BLUE detected, and what was missed. This is both a backend enhancement (Forensics agent analysis) and a frontend feature (dashboard visualization).

---

## DESIGN

### What the Forensics Reconstruction Shows

After each episode, the Forensics agent produces a structured reconstruction containing:

1. **RED's Actual Path** — Full traversal history (nodes visited, zones crossed, timestamps)
2. **BLUE's Suspected Path** — What Forensics _thought_ RED did (from `_operation_graph`)
3. **Accuracy Score** — Jaccard similarity between actual and suspected paths
4. **Key Misses** — Nodes RED visited that BLUE never detected
5. **False Positives** — Nodes BLUE investigated that RED never visited
6. **Trap Effectiveness** — Which traps triggered vs. which were wasted
7. **Dead Drop Forensics** — Which drops RED used, which BLUE tampered with, integrity status
8. **Timeline Overlay** — Step-by-step parallel timeline of RED actions vs BLUE responses

## IMPLEMENTATION STEPS

### Step 1: Enhanced Forensics Analysis Engine

**File: `cipher/agents/blue/forensics_analyzer.py`** (NEW)

```python
"""
Post-episode forensic analysis engine.
Produces structured crime scene reconstruction from episode data.
"""
from dataclasses import dataclass, field
from typing import Any

@dataclass
class ForensicsReconstruction:
    # Paths
    actual_red_path: list[int] = field(default_factory=list)
    suspected_red_path: list[int] = field(default_factory=list)
    path_accuracy: float = 0.0  # Jaccard similarity

    # Detection quality
    correctly_identified_nodes: list[int] = field(default_factory=list)
    missed_nodes: list[int] = field(default_factory=list)  # RED visited, BLUE missed
    false_positive_nodes: list[int] = field(default_factory=list)  # BLUE flagged, RED never there

    # Trap analysis
    traps_triggered: list[dict] = field(default_factory=list)
    traps_wasted: list[dict] = field(default_factory=list)  # Placed but never triggered
    trap_efficiency: float = 0.0  # triggered / total placed

    # Dead drop analysis
    drops_used: list[dict] = field(default_factory=list)
    drops_tampered: list[dict] = field(default_factory=list)
    drop_integrity_rate: float = 0.0

    # Timeline
    timeline: list[dict] = field(default_factory=list)  # [{step, red_action, blue_action, event}]

    # Summary
    investigation_grade: str = "C"  # A/B/C/D/F
    summary_text: str = ""

    def to_dict(self) -> dict:
        from dataclasses import asdict
        return asdict(self)


def reconstruct_crime_scene(state, episode_log: list[dict],
                             forensics_agent=None) -> ForensicsReconstruction:
    """Build a complete forensic reconstruction from episode data."""
    recon = ForensicsReconstruction()

    # 1. Extract RED's actual path
    recon.actual_red_path = list(getattr(state, "red_path_history", []))

    # 2. Extract BLUE's suspected path from forensics agent
    if forensics_agent and hasattr(forensics_agent, "_operation_graph"):
        for entry in forensics_agent._operation_graph:
            node_id = entry.get("node_id")
            if node_id is not None and node_id not in recon.suspected_red_path:
                recon.suspected_red_path.append(node_id)

    # 3. Compute path accuracy (Jaccard)
    actual_set = set(recon.actual_red_path)
    suspected_set = set(recon.suspected_red_path)
    union = actual_set | suspected_set
    intersection = actual_set & suspected_set
    recon.path_accuracy = len(intersection) / max(1, len(union))

    # 4. Detection quality
    recon.correctly_identified_nodes = sorted(intersection)
    recon.missed_nodes = sorted(actual_set - suspected_set)
    recon.false_positive_nodes = sorted(suspected_set - actual_set)

    # 5. Trap analysis
    registry = getattr(state, "trap_registry", None)
    if registry:
        for trap in getattr(registry, "active_blue_traps", []):
            trap_info = {"type": trap.trap_type.value, "node": trap.target_node}
            if trap.is_triggered:
                recon.traps_triggered.append(trap_info)
            else:
                recon.traps_wasted.append(trap_info)
        total = len(recon.traps_triggered) + len(recon.traps_wasted)
        recon.trap_efficiency = len(recon.traps_triggered) / max(1, total)

    # 6. Build timeline
    for entry in episode_log:
        step = entry.get("step", 0)
        agent_id = str(entry.get("agent_id", ""))
        action = str(entry.get("action_type", ""))
        team = "red" if agent_id.startswith("red_") else "blue" if agent_id.startswith("blue_") else "oversight"

        recon.timeline.append({
            "step": step,
            "team": team,
            "agent": agent_id,
            "action": action,
            "details": str(entry.get("payload", {}))[:100],
        })

    # 7. Grade the investigation
    acc = recon.path_accuracy
    if acc >= 0.8:
        recon.investigation_grade = "A"
    elif acc >= 0.6:
        recon.investigation_grade = "B"
    elif acc >= 0.4:
        recon.investigation_grade = "C"
    elif acc >= 0.2:
        recon.investigation_grade = "D"
    else:
        recon.investigation_grade = "F"

    recon.summary_text = (
        f"BLUE reconstructed {len(recon.correctly_identified_nodes)}/{len(actual_set)} "
        f"of RED's nodes (accuracy: {recon.path_accuracy:.0%}). "
        f"Missed {len(recon.missed_nodes)} nodes. "
        f"{len(recon.false_positive_nodes)} false positives. "
        f"Trap efficiency: {recon.trap_efficiency:.0%}. "
        f"Grade: {recon.investigation_grade}"
    )

    return recon
```
