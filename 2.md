## CORE COMPONENTS (13 total)

### 1. NetworkGraph.jsx — The Centerpiece
- **Library**: `react-force-graph-2d` (npm: `react-force-graph-2d`)
- Render all 50 nodes as circles, sized by importance (HVT = largest)
- **Zone coloring**: Perimeter=#555, General=#00bcd4, Sensitive=#ffc107, Critical=#ff1744
- **Zone boundaries**: Faint dashed circles separating zones (concentric rings)
- **RED agent**: Pulsing red dot with motion trail (last 3 nodes visited shown as fading red dots)
- **Honeypots**: Skull icon 💀, visible only when "BLUE view" toggle is on
- **Traps**: When BLUE places a trap → animated pulse ring expands from the node for 2s
- **Edges**: Thin gray lines; edges RED has traversed glow red; edges near traps glow blue
- **HVT node**: Gold star ⭐ pulsing in zone 3 — the target
- **Node hover**: Tooltip showing hostname, zone, files, services, trap status
- **RED movement animation**: Smooth 800ms transition along edge path, with a trailing glow

### 2. AgentMindPanel.jsx — "What Are They Thinking?"
This is the KILLER FEATURE. Shows each agent's REASONING in real-time:

**Data source**: The `reasoning` field from each `Action` object + the LLM's raw response
- When running `--live` or `--hybrid`, agents produce actual LLM reasoning text
- Display in chat-bubble style, color-coded by team
- Auto-scroll, typewriter effect (characters appear one by one for drama)
- Show the agent's **chosen action** in bold below the reasoning
- If agent chose an emergent action (Task 1), highlight with gold border + ✨ icon

**Implementation**:
- Poll `logs/agent_status.json` every 2 seconds (already written by `main.py`)
- ALSO: extend `_episode_runner.py` to write `logs/agent_thoughts.jsonl` — one line per agent action containing `{step, agent_id, reasoning, action_type, target}`

### Backend: Flask API Server

**File: `dashboard-react/api_server.py`**

```python
from flask import Flask, jsonify
from flask_cors import CORS
import json, csv, glob
from pathlib import Path

app = Flask(__name__)
CORS(app)
ROOT = Path(__file__).parent.parent

@app.route("/api/live-steps")
def live_steps():
    path = ROOT / "live_steps.jsonl"
    if not path.exists(): return jsonify([])
    lines = path.read_text(encoding="utf-8").strip().split("\n")
    return jsonify([json.loads(l) for l in lines[-50:]])

@app.route("/api/thoughts")
def thoughts():
    path = ROOT / "logs" / "agent_thoughts.jsonl"
    if not path.exists(): return jsonify([])
    lines = path.read_text(encoding="utf-8").strip().split("\n")
    return jsonify([json.loads(l) for l in lines[-20:]])

@app.route("/api/agent-status")
def agent_status():
    path = ROOT / "logs" / "agent_status.json"
    if path.exists(): return jsonify(json.loads(path.read_text(encoding="utf-8")))
    return jsonify({})

@app.route("/api/network-graph")
def network_graph():
    """Serve the current episode's network graph as nodes+edges JSON."""
    # Read from the latest episode trace or a pre-generated graph file
    traces = sorted(glob.glob(str(ROOT / "episode_traces" / "*.json")))
    if traces:
        trace = json.loads(Path(traces[-1]).read_text(encoding="utf-8"))
        return jsonify(trace.get("graph", {}))
    return jsonify({"nodes": [], "edges": []})

@app.route("/api/episodes")
def list_episodes():
    traces = sorted(glob.glob(str(ROOT / "episode_traces" / "*.json")))
    return jsonify([Path(t).name for t in traces])

@app.route("/api/episode/<filename>")
def get_episode(filename):
    path = ROOT / "episode_traces" / filename
    if path.exists(): return jsonify(json.loads(path.read_text(encoding="utf-8")))
    return jsonify({}), 404

@app.route("/api/rewards")
def rewards():
    path = ROOT / "rewards_log.csv"
    if not path.exists(): return jsonify([])
    with open(path, encoding="utf-8") as f:
        return jsonify(list(csv.DictReader(f)))

if __name__ == "__main__":
    app.run(port=5001, debug=False)
```

### Backend: Agent Thoughts Logger

**File: `cipher/training/_episode_runner.py`** — Add after each agent action:

```python
import json
from datetime import datetime
from pathlib import Path

_THOUGHTS_FILE = Path("logs") / "agent_thoughts.jsonl"

def _log_agent_thought(step: int, action) -> None:
    """Write agent reasoning to JSONL for live dashboard consumption."""
    try:
        _THOUGHTS_FILE.parent.mkdir(exist_ok=True)
        entry = {
            "step": step,
            "agent_id": action.agent_id,
            "team": "red" if action.agent_id.startswith("red_") else "blue",
            "reasoning": action.reasoning[:500] if action.reasoning else "",
            "action_type": action.action_type.value if hasattr(action.action_type, "value") else str(action.action_type),
            "target_node": action.target_node,
            "target_file": action.target_file,
            "timestamp": datetime.now().isoformat(),
        }
        with open(_THOUGHTS_FILE, "a", encoding="utf-8") as f:
            f.write(json.dumps(entry) + "\n")
    except Exception:
        pass
```

### Backend: Graph Topology Serializer

**File: `cipher/environment/graph_serializer.py`** (NEW)

```python
"""Serialize NetworkX graph to JSON for React dashboard."""
import json
from pathlib import Path

def serialize_graph_for_dashboard(graph, output_path: str = "logs/network_graph.json"):
    """Convert NetworkX graph to {nodes: [...], edges: [...]} JSON."""
    nodes = []
    for node_id, data in graph.nodes(data=True):
        zone = data.get("zone")
        zone_val = zone.value if hasattr(zone, "value") else int(zone) if zone is not None else 0
        node_type = data.get("type")
        type_val = node_type.value if hasattr(node_type, "value") else str(node_type) if node_type else "server"
        nodes.append({
            "id": node_id,
            "hostname": data.get("hostname", f"node_{node_id}"),
            "zone": zone_val,
            "type": type_val,
            "files": data.get("files", []),
            "services": data.get("services", []),
            "is_honeypot": data.get("is_honeypot", False),
            "is_entry": data.get("is_entry", False),
            "is_hvt": data.get("is_hvt", False),
        })

    edges = []
    for u, v, data in graph.edges(data=True):
        edges.append({
            "source": u,
            "target": v,
            "protocol": data.get("protocol", "tcp"),
            "bandwidth": data.get("bandwidth", 100),
        })

    result = {"nodes": nodes, "edges": edges}
    Path(output_path).parent.mkdir(exist_ok=True)
    Path(output_path).write_text(json.dumps(result, indent=2), encoding="utf-8")
    return result
```
